Notice: `scrapeClassicHistograms` corresponds to the `always_scrape_classic_histograms` field in the Prometheus configuration.
spec.scrapeNativeHistograms <boolean>	# scrapeNativeHistograms defines whether to enable scraping of native histograms. It requires Prometheus >= v3.8.0.
spec.scrapeProtocols <[]string>	# scrapeProtocols defines the protocols to negotiate during a scrape. It tells clients the protocols supported by Prometheus in order of preference (from most to least preferred). If unset, Prometheus uses its default value. It requires Prometheus >= v2.49.0.
spec.selector <object> (required)	# selector defines the label selector to select the Kubernetes `Endpoints` objects to scrape metrics from.
spec.selector.matchExpressions <[]object>	# matchExpressions is a list of label selector requirements. The requirements are ANDed.
spec.selector.matchExpressions[].key <string> (required)	# key is the label key that the selector applies to.
spec.selector.matchExpressions[].operator <string> (required)	# operator represents a key's relationship to a set of values. Valid operators are In, NotIn, Exists and DoesNotExist.
spec.selector.matchExpressions[].values <[]string>	# values is an array of string values. If the operator is In or NotIn, the values array must be non-empty. If the operator is Exists or DoesNotExist, the values array must be empty. This array is replaced during a strategic merge patch.
spec.selector.matchLabels <map[string]string>	# matchLabels is a map of {key,value} pairs. A single {key,value} in the matchLabels map is equivalent to an element of matchExpressions, whose key field is "key", the operator is "In", and the values array contains only "value". The requirements are ANDed.
spec.selectorMechanism <string> enum=RelabelConfig|RoleSelector	# selectorMechanism defines the mechanism used to select the endpoints to scrape. By default, the selection process relies on relabel configurations to filter the discovered targets. Alternatively, you can opt in for role selectors, which may offer better efficiency in large clusters. Which strategy is best for your use case needs to be carefully evaluated. It requires Prometheus >= v2.17.0.
spec.serviceDiscoveryRole <string> enum=Endpoints|EndpointSlice	# serviceDiscoveryRole defines the service discovery role used to discover targets. If set, the value should be either "Endpoints" or "EndpointSlice". Otherwise it defaults to the value defined in the Prometheus/PrometheusAgent resource.
spec.targetLabels <[]string>	# targetLabels defines the labels which are transferred from the associated Kubernetes `Service` object onto the ingested metrics.
spec.targetLimit <integer> format=int64	# targetLimit defines a limit on the number of scraped targets that will be accepted.
status <object>	# status defines the status subresource. It is under active development and is updated only when the "StatusForConfigurationResources" feature gate is enabled. Most recent observed status of the ServiceMonitor. Read-only. More info: https://github.com/kubernetes/community/blob/master/contributors/devel/sig-architecture/api-conventions.md#spec-and-status
status.bindings <[]object>	# bindings defines the list of workload resources (Prometheus, PrometheusAgent, ThanosRuler or Alertmanager) which select the configuration resource.
status.bindings[].conditions <[]object>	# conditions defines the current state of the configuration resource when bound to the referenced Workload object.
status.bindings[].conditions[].lastTransitionTime <string> (required) format=date-time	# lastTransitionTime defines the time of the last update to the current status property.
status.bindings[].conditions[].message <string>	# message defines the human-readable message indicating details for the condition's last transition.
status.bindings[].conditions[].observedGeneration <integer> format=int64	# observedGeneration defines the .metadata.generation that the condition was set based upon. For instance, if `.metadata.generation` is currently 12, but the `.status.conditions[].observedGeneration` is 9, the condition is out of date with respect to the current state of the object.
status.bindings[].conditions[].reason <string>	# reason for the condition's last transition.
status.bindings[].conditions[].status <string> (required) min=1	# status of the condition.
status.bindings[].conditions[].type <string> (required) enum=Accepted min=1	# type of the condition being reported. Currently, only "Accepted" is supported.
status.bindings[].group <string> (required) enum=monitoring.coreos.com	# group defines the group of the referenced resource.
status.bindings[].name <string> (required) min=1	# name defines the name of the referenced object.
status.bindings[].namespace <string> (required) min=1	# namespace defines the namespace of the referenced object.
status.bindings[].resource <string> (required) enum=prometheuses|prometheusagents|thanosrulers|alertmanagers	# resource defines the type of resource being referenced (e.g. Prometheus, PrometheusAgent, ThanosRuler or Alertmanager).
